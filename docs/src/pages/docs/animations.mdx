## Usage

Every animation name is passed to the `data-anim` attribute:

<CodeBlock
  code={'<div\n  data-anim="fadeInUp"\n  data-anim-duration="800ms"\n>\n  This element fades in from below over 800ms\n</div>'}
  lang="html"
/>

See [Attributes](/docs/attributes) for all customization options, or try them in the [Playground](/playground).

<script>{`
  const target = document.getElementById('anim-target');
  const label = document.getElementById('anim-current-name');
  const code = document.getElementById('anim-code-output');
  const copyBtn = document.getElementById('anim-copy-btn');
  const btns = document.querySelectorAll('.anim-btn[data-play]');
  let currentCode = '<div data-anim="fadeInUp">...</div>';

  function playAnim(name) {
    if (!target) return;
    target.style.animation = 'none';
    target.offsetWidth;
    target.style.animation = 'da-' + name + ' 1s ease both';
    if (label) label.textContent = name;
    currentCode = '<div data-anim="' + name + '">...</div>';
    if (code) code.textContent = currentCode;
    target.addEventListener('animationend', function() {
      target.style.animation = '';
    }, { once: true });
  }

  btns.forEach(function(btn) {
    btn.addEventListener('click', function() {
      btns.forEach(function(b) { b.classList.remove('active'); });
      btn.classList.add('active');
      playAnim(btn.getAttribute('data-play'));
    });
  });

  if (copyBtn) {
    copyBtn.addEventListener('click', async function() {
      await navigator.clipboard.writeText(currentCode);
      copyBtn.innerHTML = '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="20 6 9 17 4 12"/></svg>';
      setTimeout(function() {
        copyBtn.innerHTML = '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="9" y="9" width="13" height="13" rx="2" ry="2"/><path d="M5 15H4a2 2 0 01-2-2V4a2 2 0 012-2h9a2 2 0 012 2v1"/></svg>';
      }, 1500);
    });
  }
`}</script>
